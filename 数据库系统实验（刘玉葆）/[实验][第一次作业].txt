1.
CREATE TABLE CUSTOMERS
(
CID CHAR(8) UNIQUE,
CNAME CHAR(8),
CITY CHAR(8),
DISCNT FLOAT,
PRIMARY KEY(CID)
)

CREATE TABLE AGENTS
(
AID CHAR(8) UNIQUE,
ANAME CHAR(8),
CITY CHAR(8),
PERCENTS FLOAT,
PRIMARY KEY(AID)
)

CREATE TABLE PRODUCTS
(
PID CHAR(8) UNIQUE,
PNAME CHAR(8),
PRIMARY KEY(PID)
) 

2.
CREATE TABLE ORDERS
(
ORDNA CHAR(8) UNIQUE,
MONTH CHAR(8),
CID CHAR(8),
AID CHAR(8),
PID CHAR(8),
QTY CHAR(8),
DOLLARS FLOAT,
PRIMARY KEY(ORDNA),
FOREIGN KEY(CID) REFERENCES CUSTOMERS ON DELETE CASCADE,
FOREIGN KEY(AID) REFERENCES AGENTS ON DELETE CASCADE,
FOREIGN KEY(PID) REFERENCES PRODUCTS ON DELETE CASCADE
)

3.
ALTER TABLE PRODUCTS ADD CITY CHAR(8), QUANTITY CHAR(8), PRICE FLOAT

4.
CREATE INDEX CUSTOMERS_INDEX ON CUSTOMERS(CID)
CREATE INDEX AGENTS_INDEX ON AGENTS(AID)
CREATE INDEX PRODUCTS_INDEX ON PRODUCTS(PID)
CREATE INDEX ORDERS_INDEX ON ORDERS(ORDNA)

5.
DROP INDEX CUSTOMERS.CUSTOMERS_INDEX
DROP INDEX AGENTS.AGENTS_INDEX
DROP INDEX PRODUCTS.PRODUCTS_INDEX
DROP INDEX ORDERS.ORDERS_INDEX
